def feature_imp(df,model):
    feat = pd.DataFrame(columns=['feature', 'importance']) 
    feat["feature"] = df.columns
    feat["importance"] = model.best_estimator_.feature_importances_ 
    return feat.sort_values(by="importance", ascending=False)

def feature_imp_svm(df, model):
    feat = pd.DataFrame(columns=['feature', 'coefficients']) 
    feat["feature"] = df.columns
    feat["coefficients"] = model.best_estimator_.coef_[0]
    feat['coefficients_abs'] = np.abs(feat['coefficients'])
    return feat.sort_values(by="coefficients_abs", ascending=False)
